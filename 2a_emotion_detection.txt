import requests

def emotion_detector(text_to_analyze):
    """
    Runs emotion detection on the provided text using Watson NLP EmotionPredict function.
    
    Args:
        text_to_analyze (str): Text to analyze for emotions.
    
    Returns:
        dict: The 'text' attribute of the response from Watson NLP.
    """
    # Watson NLP URL and headers
    url = 'https://sn-watson-emotion.labs.skills.network/v1/watson.runtime.nlp.v1/NlpService/EmotionPredict'
    headers = {"grpc-metadata-mm-model-id": "emotion_aggregated-workflow_lang_en_stock"}
    
    # Prepare JSON input
    input_json = {
        "raw_document": {
            "text": text_to_analyze
        }
    }
    
    # Send POST request
    response = requests.post(url, headers=headers, json=input_json)
    
    # Return the 'text' attribute of the response JSON
    return response.json().get('text')
